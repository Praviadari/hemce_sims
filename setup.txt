/c/Windows/System32/WindowsPowerShell/v1.0/powershell -NoProfile -ExecutionPolicy Bypass -Command "\$p=Get-NetTCPConnection -State Listen -LocalPort 5173 -ErrorAction SilentlyContinue | Select-Object -First 1 -ExpandProperty OwningProcess; if(\$p){ Stop-Process -Id \$p -Force }; Start-Process -FilePath 'C:\Program Files\nodejs\node.exe' -ArgumentList 'C:\hemce_sims_local\node_modules\vite\bin\vite.js' -WorkingDirectory 'C:\hemce_sims_local' -WindowStyle Normal"
